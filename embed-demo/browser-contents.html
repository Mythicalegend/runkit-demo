Creates an array of elements, sorted in ascending order by the results of running each element in a collection through each iteratee

var users = [
  { firstName: "Fred",   lastName: "Turing" },
  { firstName: "Barney", lastName: "Johnson" },
  { firstName: "Fred",   lastName: "Michaels" }
];
 
_.sortBy(users, user => user.lastName)
